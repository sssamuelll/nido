# Events Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add time-bounded Events (viajes, ocasiones especiales) that group expenses with their own budget, category donut breakdown, and dedicated detail view — integrated into the existing category/budget flow.

**Architecture:** New `events` + `event_categories` tables. New `server/routes/events.ts` for CRUD + detail endpoint. Frontend: toggle in existing CategoryModal, event cards in Dashboard budget list, new EventDetail view at `/events/:id`, optional event dropdown in AddExpense.

**Tech Stack:** SQLite (via better-sqlite3-wrapper), Express, React, TypeScript. Reuses existing SpendingDonut component from Analytics.

**Spec:** `docs/superpowers/specs/2026-04-15-events-design.md`

```typescript
import { Router } from 'express';
import { getDatabase } from '../db.js';
import { AuthRequest } from '../auth.js';

const router = Router();

// GET /events — list events for household
router.get('/', async (req: AuthRequest, res) => {
  try {
    const db = getDatabase();
    const userId = req.user!.id;
    const context = (req.query.context as string) === 'personal' ? 'personal' : 'shared';

    const household = await db.get<{ household_id: number }>(
      'SELECT household_id FROM app_users WHERE id = ?', userId
    );
    if (!household) return res.status(404).json({ error: 'Household no encontrado' });

    const contextFilter = context === 'shared'
      ? "context = 'shared'"
      : `context = 'personal' AND owner_user_id = ?`;
    const contextParams = context === 'personal' ? [userId] : [];

    const events = await db.all(
      `SELECT e.*, COALESCE(SUM(exp.amount), 0) as total_spent
       FROM events e
       LEFT JOIN expenses exp ON exp.event_id = e.id
       WHERE e.household_id = ? AND ${contextFilter}
       GROUP BY e.id
       ORDER BY e.end_date >= date('now') DESC, e.end_date ASC`,
      household.household_id, ...contextParams
    );

    res.json(events);
  } catch (error) {
    console.error('Error listing events:', error);
    res.status(500).json({ error: 'Error al obtener eventos' });
  }
});

// GET /events/:id — full event detail with KPIs, categories, expenses
router.get('/:id', async (req: AuthRequest, res) => {
  try {
    const db = getDatabase();
    const userId = req.user!.id;
    const eventId = Number(req.params.id);

    const household = await db.get<{ household_id: number }>(
      'SELECT household_id FROM app_users WHERE id = ?', userId
    );
    if (!household) return res.status(404).json({ error: 'Household no encontrado' });

    const event = await db.get(
      'SELECT * FROM events WHERE id = ? AND household_id = ?',
      eventId, household.household_id
    );
    if (!event) return res.status(404).json({ error: 'Evento no encontrado' });

    // KPIs
    const spentRow = await db.get<{ total: number }>(
      'SELECT COALESCE(SUM(amount), 0) as total FROM expenses WHERE event_id = ?', eventId
    );
    const spent = spentRow?.total ?? 0;
    const kpis = {
      budget: (event as any).budget_amount,
      spent,
      remaining: (event as any).budget_amount - spent,
    };

    // Category breakdown
    const categoryRows = await db.all<{ name: string; amount: number }[]>(
      `SELECT e.category as name, SUM(e.amount) as amount
       FROM expenses e WHERE e.event_id = ?
       GROUP BY e.category ORDER BY amount DESC`,
      eventId
    );

    // Get emoji/color from categories table or event_categories
    const householdId = household.household_id;
    const catMeta = await db.all<{ name: string; emoji: string; color: string }[]>(
      'SELECT name, emoji, color FROM categories WHERE household_id = ?', householdId
    );
    const eventCatMeta = await db.all<{ name: string; emoji: string; color: string }[]>(
      'SELECT name, emoji, color FROM event_categories WHERE event_id = ?', eventId
    );
    const metaMap: Record<string, { emoji: string; color: string }> = {};
    for (const c of catMeta) metaMap[c.name] = { emoji: c.emoji, color: c.color };
    for (const c of eventCatMeta) metaMap[c.name] = { emoji: c.emoji, color: c.color };

    const totalSpent = categoryRows.reduce((s, r) => s + r.amount, 0);
    const categories = categoryRows.map(row => ({
      name: row.name,
      emoji: metaMap[row.name]?.emoji ?? '📦',
      color: metaMap[row.name]?.color ?? '#888888',
      amount: row.amount,
      pct: totalSpent > 0 ? Math.round((row.amount / totalSpent) * 100) : 0,
    }));

    // Expenses
    const expenses = await db.all(
      `SELECT id, description, amount, category, date, paid_by, type
       FROM expenses WHERE event_id = ? ORDER BY date DESC, id DESC`,
      eventId
    );

    res.json({ event, kpis, categories, expenses });
  } catch (error) {
    console.error('Error fetching event detail:', error);
    res.status(500).json({ error: 'Error al obtener detalle del evento' });
  }
});

// POST /events — create event
router.post('/', async (req: AuthRequest, res) => {
  try {
    const db = getDatabase();
    const userId = req.user!.id;
    const { name, emoji, budget_amount, start_date, end_date, goal_id, context, subcategories } = req.body;

    if (!name || !start_date || !end_date) {
      return res.status(400).json({ error: 'Nombre, fecha inicio y fecha fin son requeridos' });
    }

    const household = await db.get<{ household_id: number }>(
      'SELECT household_id FROM app_users WHERE id = ?', userId
    );
    if (!household) return res.status(404).json({ error: 'Household no encontrado' });

    const result = await db.run(
      `INSERT INTO events (household_id, name, emoji, budget_amount, start_date, end_date, goal_id, context, owner_user_id, created_by)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      household.household_id,
      name,
      emoji || '✈️',
      budget_amount || 0,
      start_date,
      end_date,
      goal_id || null,
      context || 'shared',
      context === 'personal' ? userId : null,
      userId
    );

    // Create event-specific subcategories if provided
    if (Array.isArray(subcategories)) {
      for (const sub of subcategories) {
        if (sub.name && sub.emoji && sub.color) {
          await db.run(
            'INSERT OR IGNORE INTO event_categories (event_id, name, emoji, color) VALUES (?, ?, ?, ?)',
            result.lastID, sub.name, sub.emoji, sub.color
          );
        }
      }
    }

    const event = await db.get('SELECT * FROM events WHERE id = ?', result.lastID);
    res.status(201).json(event);
  } catch (error) {
    console.error('Error creating event:', error);
    res.status(500).json({ error: 'Error al crear evento' });
  }
});

// PUT /events/:id — update event
router.put('/:id', async (req: AuthRequest, res) => {
  try {
    const db = getDatabase();
    const userId = req.user!.id;
    const eventId = Number(req.params.id);
    const { name, emoji, budget_amount, start_date, end_date, goal_id } = req.body;

    const household = await db.get<{ household_id: number }>(
      'SELECT household_id FROM app_users WHERE id = ?', userId
    );
    if (!household) return res.status(404).json({ error: 'Household no encontrado' });

    const existing = await db.get(
      'SELECT * FROM events WHERE id = ? AND household_id = ?',
      eventId, household.household_id
    );
    if (!existing) return res.status(404).json({ error: 'Evento no encontrado' });

    await db.run(
      `UPDATE events SET
        name = COALESCE(?, name),
        emoji = COALESCE(?, emoji),
        budget_amount = COALESCE(?, budget_amount),
        start_date = COALESCE(?, start_date),
        end_date = COALESCE(?, end_date),
        goal_id = ?
       WHERE id = ?`,
      name, emoji, budget_amount, start_date, end_date,
      goal_id !== undefined ? goal_id : (existing as any).goal_id,
      eventId
    );

    const updated = await db.get('SELECT * FROM events WHERE id = ?', eventId);
    res.json(updated);
  } catch (error) {
    console.error('Error updating event:', error);
    res.status(500).json({ error: 'Error al actualizar evento' });
  }
});

// DELETE /events/:id — delete event (expenses keep, lose event_id)
router.delete('/:id', async (req: AuthRequest, res) => {
  try {
    const db = getDatabase();
    const userId = req.user!.id;
    const eventId = Number(req.params.id);

    const household = await db.get<{ household_id: number }>(
      'SELECT household_id FROM app_users WHERE id = ?', userId
    );
    if (!household) return res.status(404).json({ error: 'Household no encontrado' });

    const existing = await db.get(
      'SELECT * FROM events WHERE id = ? AND household_id = ?',
      eventId, household.household_id
    );
    if (!existing) return res.status(404).json({ error: 'Evento no encontrado' });

    await db.run('UPDATE expenses SET event_id = NULL WHERE event_id = ?', eventId);
    await db.run('DELETE FROM events WHERE id = ?', eventId);

    res.status(204).send();
  } catch (error) {
    console.error('Error deleting event:', error);
    res.status(500).json({ error: 'Error al eliminar evento' });
  }
});

export default router;
```

  paid_by: string;

// Reuse SpendingDonut inline — same SVG donut logic as Analytics
const EventDonut: React.FC<{ categories: EventCategory[] }> = ({ categories }) => {
  const total = categories.reduce((s, c) => s + c.amount, 0);
  if (total === 0) return null;

  const size = 200;
  const cx = size / 2, cy = size / 2, r = 70;
  const circumference = 2 * Math.PI * r;
  let offset = 0;

  return (
    <div className="ev-donut-section">
      <svg viewBox={`0 0 ${size} ${size}`} className="ev-donut-svg">
        {categories.map((cat, i) => {
          const pct = cat.amount / total;
          const dash = pct * circumference;
          const gap = circumference - dash;
          const currentOffset = offset;
          offset += dash;
          return (
            <circle
              key={i}
              cx={cx} cy={cy} r={r}
              fill="none" stroke={cat.color} strokeWidth="24"
              strokeDasharray={`${dash} ${gap}`}
              strokeDashoffset={-currentOffset}
              style={{ transform: 'rotate(-90deg)', transformOrigin: 'center' }}
            />
          );
        })}
        <text x={cx} y={cy - 6} textAnchor="middle" fill="var(--text)" fontSize="20" fontWeight="700">
          €{total.toLocaleString('es-ES')}
        </text>
        <text x={cx} y={cy + 14} textAnchor="middle" fill="var(--ts)" fontSize="11">
          gastado
        </text>
      </svg>
      <div className="ev-donut-legend">
        {categories.map((cat, i) => (
          <div key={i} className="ev-donut-legend__item">
            <span className="ev-donut-legend__emoji">{cat.emoji}</span>
            <span className="ev-donut-legend__name">{cat.name}:</span>
            <span className="ev-donut-legend__amount">€{cat.amount.toLocaleString('es-ES')}</span>
            <span className="ev-donut-legend__pct">({cat.pct}%)</span>
            <div className="ev-donut-legend__bar" style={{ '--bar-color': cat.color } as React.CSSProperties} />
          </div>
        ))}
      </div>
    </div>
  );
};

export const EventDetail: React.FC = () => {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const [data, setData] = useState<any>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const load = async () => {
      try {
        const detail = await Api.getEventDetail(Number(id));
        setData(detail);
      } catch (err) {
        console.error('Error loading event:', err);
      } finally {
        setLoading(false);
      }
    };
    void load();
  }, [id]);

  if (loading) {
    return (
      <div className="loading-screen">
        <div className="loading-screen__logo"><span>N</span></div>
        <div className="loading-screen__text">Cargando...</div>
      </div>
    );
  }

  if (!data) {
    return (
      <div className="error-view">
        <div className="error-view__msg">Evento no encontrado</div>
      </div>
    );
  }

  const { event, kpis, categories, expenses } = data;
  const pctUsed = kpis.budget > 0 ? Math.round((kpis.spent / kpis.budget) * 100) : 0;

  // Group expenses by date
  const grouped: Record<string, EventExpense[]> = {};
  for (const exp of expenses) {
    if (!grouped[exp.date]) grouped[exp.date] = [];
    grouped[exp.date].push(exp);
  }
  const dateGroups = Object.entries(grouped).sort(([a], [b]) => b.localeCompare(a));

  const formatDateLabel = (dateStr: string) => {
    const d = new Date(dateStr + 'T12:00:00');
    const days = ['DOM', 'LUN', 'MAR', 'MIÉ', 'JUE', 'VIE', 'SÁB'];
    return `${days[d.getDay()]} ${d.getDate()}`;
  };

  return (
    <>
      <div className="topbar">
        <button className="ev-back" onClick={() => navigate('/')}>
          <ChevronLeft size={18} /> Volver al dashboard
        </button>
      </div>

      <h1 className="ev-title">{event.emoji} {event.name}</h1>

      <div className="ev-kpis">
        <div className="card ev-kpi">
          <div className="ev-kpi__label">Presupuesto Total</div>
          <div className="ev-kpi__value">€{kpis.budget.toLocaleString('es-ES')}</div>
          <div className="ev-kpi__bar">
            <div className="ev-kpi__bar-fill ev-kpi__bar-fill--neutral" style={{ width: `${pctUsed}%` }} />
          </div>
          <div className="ev-kpi__sub">{pctUsed}%</div>
        </div>
        <div className="card ev-kpi">
          <div className="ev-kpi__label">Gastado</div>
          <div className="ev-kpi__value" style={{ color: 'var(--green)' }}>€{kpis.spent.toLocaleString('es-ES')}</div>
          <div className="ev-kpi__bar">
            <div className="ev-kpi__bar-fill ev-kpi__bar-fill--green" style={{ width: `${pctUsed}%` }} />
          </div>
        </div>
        <div className="card ev-kpi">
          <div className="ev-kpi__label">Restante</div>
          <div className="ev-kpi__value" style={{ color: kpis.remaining < 0 ? 'var(--red)' : 'var(--orange)' }}>
            €{Math.abs(kpis.remaining).toLocaleString('es-ES')}
          </div>
          <div className="ev-kpi__bar">
            <div className="ev-kpi__bar-fill ev-kpi__bar-fill--orange" style={{ width: `${Math.max(0, 100 - pctUsed)}%` }} />
          </div>
        </div>
      </div>

      {categories.length > 0 && (
        <div className="card ev-breakdown">
          <EventDonut categories={categories} />
        </div>
      )}

      <div className="card ev-transactions">
        <h3>Transacciones del evento</h3>
        {dateGroups.length === 0 && (
          <div className="empty-view">
            <div className="empty-view__text">No hay gastos registrados en este evento</div>
          </div>
        )}
        {dateGroups.map(([date, exps]) => (
          <div key={date}>
            <div className="ev-date-label">{formatDateLabel(date)}</div>
            {exps.map(exp => (
              <div key={exp.id} className="ev-expense-row">
                <div className="ev-expense-row__left">
                  <div className="ev-expense-row__desc">{exp.description}</div>
                  <div className="ev-expense-row__cat">{exp.category}</div>
                </div>
                <div className="ev-expense-row__right">
                  <div className="ev-expense-row__amount">−€{exp.amount.toFixed(2)}</div>
                  <div className="ev-expense-row__who">{exp.paid_by}</div>
                </div>
              </div>
            ))}
          </div>
        ))}
      </div>
    </>
  );
};
